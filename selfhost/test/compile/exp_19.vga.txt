label main
  push bp
  cp sp bp

  # 関数の処理本体
  sub_sp 1
  cp 0 reg_a
  cp reg_a [bp:-1]

label while_1
  cp [bp:-1] reg_a
  push reg_a
  cp 0 reg_a
  push reg_a
  pop reg_b
  pop reg_a
  compare
  jump_eq then_2
  cp 0 reg_a
  jump end_eq_2
label then_2
  cp 1 reg_a
label end_eq_2
  cp 1 reg_b
  compare
  jump_eq true_1
  jump end_while_1
label true_1
  cp 1 reg_a
  cp reg_a [bp:-1]
  jump while_1
label end_while_1


  cp bp sp
  pop bp
  ret
